      ```javascript
      function calculateMA(data, period) {
        const result = [];
        for (let i = 0; i < data.length; i++) {
          if (i < period - 1) {
            result.push(null);
          } else {
            let sum = 0;
            for (let j = 0; j < period; j++) {
              sum += data[i - j].close;
            }
            result.push(+(sum / period).toFixed(2));
          }
        }
        return result;
      }
      ```
